def convert(c, n):
    converted = ord(c) + n%26
    if (c.islower() and not chr(converted).islower()) or (c.isupper() and not chr(converted).isupper()):
        converted -= 26
    return chr(converted)

def caesar(s, n):
    return ''.join(convert(c, n) if c.isalpha() else c for c in s)
